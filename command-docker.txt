#Comandos varios
docker inspect container -> ver la configuracion del contenedro salida json
docker port container -> lista los puertos expuestos del contenedor

#crear contenedor y iniciar en base a una imagen  usuario especifico  -u 
#tener en cuenta --privileged y /usr/sbin/init procesos de systemctl
docker run --privileged --name prueba-centos -d -it --memory=1000m -p 8081:80 centos:7 /usr/sbin/init

#comando para generar imagen en base a un dockerfile o url
docker build -t laboratorio-docker:v1 url o .

#comandos networking
docker network ls ---> listar tipos de redes 
docker network inspect bridge -> inspecionar red existente
docker network create --subnet 10.130.2.0/24 --gateway 10.130.1.254 red1 -> crear red bridge definiendo un rango de ips
docker network connect red1 container -> vincular red a un contenedor existente
docker run --name red2 -itd -h myhost2 --network red1 centos:7
